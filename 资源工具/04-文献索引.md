#### 自定义检索脚本

```python
# 文献自动检索脚本示例
import scholarly
import pandas as pd
from datetime import datetime

def search_literature(keywords, max_results=100):
    """
    自动检索相关文献
    """
    results = []
    
    for keyword in keywords:
        search_query = scholarly.search_pubs(keyword)
        
        for i, pub in enumerate(search_query):
            if i >= max_results:
                break
                
            result = {
                'title': pub.bib.get('title', ''),
                'authors': pub.bib.get('author', ''),
                'year': pub.bib.get('year', ''),
                'venue': pub.bib.get('venue', ''),
                'citations': pub.citedby,
                'url': pub.bib.get('url', ''),
                'abstract': pub.bib.get('abstract', ''),
                'keywords': keyword,
                'retrieved_date': datetime.now().strftime('%Y-%m-%d')
            }
            results.append(result)
    
    # 保存到CSV
    df = pd.DataFrame(results)
    df.to_csv('literature_search_results.csv', index=False, encoding='utf-8-sig')
    
    return df

# 使用示例
keywords = [
    'cislunar trajectory design',
    'lunar gateway orbit',
    'weak stability boundary',
    'invariant manifold lunar'
]

df_results = search_literature(keywords, max_results=50)
print(f"检索到 {len(df_results)} 篇文献")
```
